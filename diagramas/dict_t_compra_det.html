Tabla: Compra
CampoTipo de DatoLong.DescripciónRestricciones
id_compraINT—Identificador único de la orden de compra.PK, NOT NULL, AUTO_INCREMENT
id_proveedorINT—Proveedor que realizó el despacho de mercancía.FK → Proveedor(id_proveedor), NOT NULL
id_usuarioINT—Administrador o empleado que registró la compra.FK → Usuario(id_usuario), NOT NULL
numero_facturaVARCHAR50Número de factura físico del proveedor.NULL
fecha_compraDATETIME—Fecha y hora exacta en que se realizó la compra.NOT NULL, DEFAULT NOW()
totalDECIMAL10,2Monto total pagado. Suma de subtotales.NOT NULL
metodo_pagoENUM—Forma de pago: 'efectivo', 'transferencia' o 'credito'.NOT NULL
